% symmetric.tex

\documentclass{standalone}

\usepackage{tikz}
\usetikzlibrary{arrows.meta, positioning}

\begin{document}
\begin{tikzpicture}[every node/.style = {draw, circle, minimum size = 10pt},
  relation/.style = {>=Stealth, ->, thick}]
  \node (a) [] {$a$};
  \node (b) [right = 1.5cm of a] {$b$};

  \draw [relation, bend left] (a) to (b);
  \draw [relation, bend left] (b) to (a);
\end{tikzpicture}
\end{document}
