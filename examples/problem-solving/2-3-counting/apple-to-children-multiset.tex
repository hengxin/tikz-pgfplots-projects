\documentclass{standalone}
\usepackage{tikz}
\usepackage{tikz-qtree}
\usetikzlibrary{shapes, positioning, arrows.meta}

\begin{document}
\begin{tikzpicture}[pass/.style = {draw, ->, >=Stealth, line width = 3pt, blue, outer sep = 0}]
  % apples
  \foreach \id in {1, ..., 7} {
    \node (apple\id) [inner sep = 0] at (\id * 3.5, 0) {\includegraphics[scale = 0.20]{figs/apple-logo}};
  }

  % children
  \node (children) [below = 3.0cm of apple4, inner sep = 0] {\includegraphics[scale = 0.60]{figs/5-children}};
  % children names
  \foreach \id/\anchor in {1/160, 2/145, 3/90, 4/35, 5/20} {
    \node (child\id) [draw, below = 4.5cm of children.\anchor, font = \Huge, red, circle] {$\id$};
  }
  
  % pass out the apples to children
  \foreach \id/\anchor in {1/160, 2/160, 3/90, 4/160, 5/35, 6/90, 7/20} {
    \draw[pass] (apple\id.south) to (children.\anchor);
  }
\end{tikzpicture}
\end{document}
