\documentclass[tikz]{standalone}
\usetikzlibrary{shapes, positioning, decorations.pathmorphing}

\begin{document}
\begin{tikzpicture}[every node/.style = {draw, circle, minimum size = 8pt, font = \large},
    node distance = 1.5cm,
    every edge/.style = {draw, ->},
    path/.style = {->, decorate, decoration = {snake, amplitude = .4mm, segment length = 2mm, post length = 1mm}}]
  \node (s) [red] {$s$};

  \node (x) [right = of s, label = {[above = -6pt]: $< 0$}] {$x$};
  \node (t) [right = of x, blue, label = {[above = -6pt]: $-\infty$}] {$t$};

  \path (s) edge[path] (x)
	(x) edge[path, out = 45, in = 135, looseness = 8] (x)
	    edge[path] (t);
\end{tikzpicture}
\end{document}
