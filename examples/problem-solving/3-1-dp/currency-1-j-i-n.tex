\documentclass[tikz]{standalone}
\usetikzlibrary{arrows.meta, positioning, decorations.pathmorphing}

\tikzset{
  zz/.style = {decorate, decoration = {
    zigzag,
    segment length = 5,
    amplitude = 1.5,
    post = lineto,
    post length = 3pt}}
}

\begin{document}
\begin{tikzpicture}[every edge/.style = {draw, ->, zz}]
  \node (1) [draw, circle] {$1$};
  \node (j) [draw, circle, below right = 0.40cm and 1.0cm of 1, teal] {$j$};
  \node (i) [draw, circle, above right = 0.40cm and 1.0cm of j, red] {$i$};
  \node (n) [draw, circle, right = of i] {$n$};

  \path (1) edge (i)
	(i) edge (n)
	(1) edge[teal] (j)
	(j) edge[teal] (i)
	(j) edge[bend right = 30, blue] (n)
	(i.north) edge[blue, bend right = 60] (j.north);
\end{tikzpicture}
\end{document}
