\documentclass[tikz]{standalone}
\usetikzlibrary{positioning, shapes, fit}

\begin{document}
\begin{tikzpicture}[every node/.style = {draw, circle, inner sep = 2pt},
    comp/.style = {yshift = -8pt, draw, thick, dashed, teal, ellipse, minimum height = 50pt, minimum width = 25pt}]
  \node (u) {$u$};
  \node (v) [right = of u] {$v$};

  \node (x) [fit = (u), comp, label = {[above, teal] $X$}] {};
  \node (y) [fit = (v), comp, label = {[above, teal] $Y$}] {};

  \path (u) edge (v);
\end{tikzpicture}
\end{document}
