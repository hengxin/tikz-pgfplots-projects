% file: graph-paths/most-critical-edge.tex

\documentclass[tikz]{standalone}

\usetikzlibrary{positioning, decorations.pathmorphing}

\begin{document}
\begin{tikzpicture}[n/.style = {circle, draw}, % node
    leadsto/.style = {->, decorate, decoration = {
    	zigzag, amplitude = 1.50pt, segment length = 1.5mm, pre length = 2.5pt, post length = 2.5pt}},
    e/.style = {->}, % edge
  ]

  \foreach \name/\pos/\label in {s/0/s, x/1/, y/2/, z/4/, a/5/, t/6/t} {
    \node (\name) [n] at (\pos cm, 0) {$\label$};
  }

  \path (s) edge[e] (x)
	(x) edge[e] (y)
	(y) edge[leadsto, bend left] (z)
	    edge[leadsto, bend right] (z)
	(z) edge[e] (a)
	(a) edge[e] (t);
\end{tikzpicture}
\end{document}
