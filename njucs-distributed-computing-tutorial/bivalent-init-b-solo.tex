  \draw [decorate, blue, very thick, decoration = {brace, mirror, amplitude = 10pt, raise = 5pt}] 
    (b3.west) -- (a2.west) node [midway, sloped, above = 16pt, font = \Large] {$A$ runs \emph{solo}};
