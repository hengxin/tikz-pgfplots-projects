\begin{document}
\begin{tikzpicture}[

  \node (rx1) [wop, right = 1.5cm of wx0] {$r(x)1$};

  \node (p1) [process, below = 1.5cm of p0] {$p_1:$};

  \rw{wx0}{rx0}
