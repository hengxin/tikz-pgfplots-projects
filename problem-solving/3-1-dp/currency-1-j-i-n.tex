\documentclass[tikz]{standalone}
\usetikzlibrary{arrows.meta, positioning, decorations.pathmorphing}

\tikzset{
  zz/.style = {decorate, decoration = {
    zigzag,
    segment length = 5,
    amplitude = 1.5,
    post = lineto,
    post length = 3pt}}
}

\begin{document}
\begin{tikzpicture}[every edge/.style = {draw, ->, zz}]
  \node (1) [draw, circle] {$1$};
  \node (j) [draw, circle, right = of 1, red] {$j$};
  \node (i) [draw, circle, right = of j, red] {$i$};
  \node (n) [draw, circle, right = of i] {$n$};

  \path (1) edge (j)
	(j) edge[blue] (i)
	(i) edge (n)
	(i.north) edge[blue, bend right = 45] (j.north);
\end{tikzpicture}
\end{document}
