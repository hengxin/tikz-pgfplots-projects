\documentclass[tikz]{standalone}
\usetikzlibrary{positioning}

\begin{document}
\begin{tikzpicture}[vertex/.style = {draw, circle, minimum size = 8pt, font = \large},
    node distance = 1.0 and 2.5cm,
    every edge/.style = {draw, -}]

  \node (v) [vertex] {$v$};

  \node (v') [vertex, above right = of v] {$v'$};
  \node (v'') [vertex, below right = of v] {$v''$};

  \draw (v') to node [right = 5pt] {$0$} (v'');
\end{tikzpicture}
\end{document}
