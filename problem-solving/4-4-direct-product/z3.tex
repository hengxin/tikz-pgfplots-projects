% z3.tex

\documentclass[tikz]{standalone}
\usetikzlibrary{positioning}

\begin{document}
\begin{tikzpicture}[ele/.style = {circle, minimum size = 8pt, fill = lightgray},
  op/.style = {->, blue, very thick}]
  \node (0) [ele] {$0$};
  \node (1) [ele, right = of 0] {$1$};
  \node (2) [ele, right = of 1] {$2$};

  \path (0) edge[op] (1)
		(1) edge[op] (2)
		(2) edge[op, bend right] (0);
\end{tikzpicture}
\end{document}
