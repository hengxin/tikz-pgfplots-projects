\documentclass[tikz]{standalone}
\usepackage{xifthen}

\begin{document}
\begin{tikzpicture}[every node/.style = {draw, circle, minimum size = 10pt}]
  \def\rnum{4}
  \def\cnum{5}

  \foreach \r in {1, ..., \rnum} {
    \foreach \c in {1, ..., \cnum} {
      \pgfmathparse{mod(\r + \c, 2)} 
      \let\parity\pgfmathresult 

      % \ifthenelse{\NOT \isodd{\r + \c}}
      \ifthenelse{\equal{\parity}{0.0}}
	{\node (\c\r) [fill = lightgray] at (\c, \r){}}
	{\node (\c\r) at (\c, \r)};
    }
  }

  \path (21) edge (13)
  	     edge (33)
	     edge (42);
  \path (41) edge (22)
  	     edge (33)
	     edge (53);
  \path (14) edge (22)
  	     edge (33);
  \path (34) edge (13)
  	     edge (22)
	     edge (42)
	     edge (53);
  \path (54) edge (33)
	     edge (42);
\end{tikzpicture}
