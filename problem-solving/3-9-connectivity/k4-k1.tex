\documentclass[tikz]{standalone}
\usetikzlibrary{positioning}

\begin{document}
\begin{tikzpicture}[every node/.style = {draw, circle, inner sep = 2pt}]
  \node (u) {};
  \node (v) [below = of u] {};

  \node (x) [right = of u] {};
  \node (y) [right = of v] {};

  \node (z) [right = 0.80cm of y] {};

  \path (u) edge (v)
  	    edge (y)
  	(v) edge (y)
	    edge (x)
	(y) edge (x)
	    edge (z)
	(x) edge (u);
\end{tikzpicture}
\end{document}
