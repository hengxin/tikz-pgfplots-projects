\tikzset{
 process/.style = {rectangle, font = \huge, text width=6em, align = right},
 put/.style = {rectangle, rounded corners, font =\Large, draw, fill = blue!30, text width=9em, text centered},
 get/.style = {rectangle, rounded corners, font =\Large, draw, fill = yellow!30, text width=9em, text centered},
}
  \node (p-c) [process] {$c:$};
  \node (o) [get, right = 0.1cm of p-c] {$o=\get(k')$};
  \node (g) [get, right = 3.0cm of o] {$g = \get(k)$};

  \node (p-j) [process, below = 1.2cm of p-c] {$j\neq c:$};
  \node (p') [put, right = 0.1cm of p-j] {$p'=\pput(k,\_)$};

  \node (p-i) [process, below = 1.2cm of p-j] {$i:$};
  \node (p) [put, right = 0.1cm of p-i] {$p=\pput(k,\_)$};

  \pva{o}{g};
  \ar{p}{p'};
  \ar{p'}{o};
  \rvb{p}{g};
