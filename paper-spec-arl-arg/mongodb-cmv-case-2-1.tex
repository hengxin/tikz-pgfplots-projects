  \node (p-j) [process, below = 0.8cm of p-c] {$j\neq c:$};
  \node (p') [put, right = 0.1cm of p-j] {$p'=\pput(k,\_)$};

  \node (p-i) [process, below = 0.8cm of p-j] {$i:$};
  \node (p) [put, right = 0.1cm of p-i] {$p=\pput(k,\_)$};

  \pbva{e}{g};
  \ar{p}{p'};
  \ar{p'}{e};
  \rvb{p}{g};
