% file: graph-paths/dijkstra-negative-edges.tex

\documentclass[tikz]{standalone}

\usetikzlibrary{positioning}

\begin{document}
\begin{tikzpicture}[n/.style = {circle, draw}, % node
    e/.style = {->}, % edge
    node distance = 1.0cm and 1.5cm,
  ]
  \node (s) [n] {$S$};

  \node (a) [n, above right = of s] {$A$};
  \node (b) [n, below right = of s] {$B$};

  \path (s) edge[e] node[above] {$3$} (a)
	    edge[e] node[below] {$4$} (b)
	(b) edge[e, red] node[right] {$-2$} (a);
\end{tikzpicture}
\end{document}
