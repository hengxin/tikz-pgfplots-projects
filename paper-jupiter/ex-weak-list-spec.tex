\documentclass{standalone}

\usepackage{tikz}
\usetikzlibrary{shapes, positioning, arrows.meta}

\begin{document}
\begin{tikzpicture}[state/.style = {draw, thick, rectangle},
  elem/.style = {draw, thick, circle},
  order/.style = {>=Stealth, ->, bend left}]
  % states
  \begin{scope}
    \node (ab) [state] {$\sigma_0: ab$};
    \node (ba) [state, right = of ab] {$\sigma_1: ba$};
  \end{scope}

  % digraph
  \begin{scope}
    \node (a) [elem, below = of ab] {$a$};
    \node (b) [elem, below = of ba] {$b$};

    \draw [order] (a) to node [sloped, above] {$\sigma_0: ab$} (b);
    \draw [order] (b) to node [sloped, below] {$\sigma_1: ba$} (a);
  \end{scope}
\end{tikzpicture}
\end{document}
