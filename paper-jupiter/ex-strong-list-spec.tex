% File: ex-strong-list-spec.tex

\documentclass{standalone}

\usepackage{tikz}
\usetikzlibrary{shapes, positioning, arrows.meta}

\begin{document}
\begin{tikzpicture}[state/.style = {draw, thick, rectangle, minimum height = 15pt},
  elem/.style = {draw, thick, circle},
  order/.style = {>=Stealth, ->}]
  % states
  \begin{scope}
    \node (ax) [state] {$\sigma_0: ax$};
    \node (xb) [state, right = 0.50cm of ax] {$\sigma_1: xb$};
    \node (ba) [state, right = 0.50cm of xb] {$\sigma_2: ba$};
  \end{scope}

  % digraph
  \begin{scope}
    \node (a) [elem, below = 1.5cm of ax] {$a$};
    \node (x) [elem, below = 0.50cm of xb] {$x$};
    \node (b) [elem, below = 1.5cm of ba] {$b$};

    \draw [order] (a) to node [sloped, above] {$\sigma_0: ax$} (x);
    \draw [order] (x) to node [sloped, above] {$\sigma_1: xb$} (b);
    \draw [order] (b) to node [sloped, below] {$\sigma_2: ba$} (a);
  \end{scope}
\end{tikzpicture}
\end{document}
